#include <stdio.h>
#include <string.h>
#define N 5

int main() {
    char str[N][50], temp[50];
    int i, j;

    printf("Enter %d strings:\n", N);
    for (i = 0; i < N; i++) {
        printf("String %d: ", i + 1);
        scanf(" %49[^\n]", str[i]);
    }

    /* Bubble sort using strcmp */
    for (i = 0; i < N - 1; i++)
        for (j = 0; j < N - 1 - i; j++)
            if (strcmp(str[j], str[j+1]) > 0) {
                strcpy(temp,    str[j]);
                strcpy(str[j],  str[j+1]);
                strcpy(str[j+1], temp);
            }

    printf("\nSorted strings (A to Z):\n");
    for (i = 0; i < N; i++)
        printf("  %s\n", str[i]);

    return 0;
}
