#include <stdio.h>
#define PI 3.14159

void areaOfCircle(float *radius, float *area) {
    *area = PI * (*radius) * (*radius);
}

int main() {
    float r, area;
    printf("Enter radius: ");
    scanf("%f", &r);

    areaOfCircle(&r, &area);

    printf("Radius = %.2f\n", r);
    printf("Area   = %.4f\n", area);

    return 0;
}
