#include <stdio.h>
#include <stdlib.h>

int isPrime(int n) {
    if (n < 2) return 0;
    int i;
    for (i = 2; i * i <= n; i++)
        if (n % i == 0) return 0;
    return 1;
}

int main() {
    int n, i, sum = 0;

    printf("Enter N: ");
    scanf("%d", &n);

    int *arr = (int *)malloc(n * sizeof(int));
    if (!arr) { printf("Alloc failed\n"); return 1; }

    printf("Enter %d elements:\n", n);
    for (i = 0; i < n; i++) scanf("%d", arr + i);

    printf("Prime numbers: ");
    for (i = 0; i < n; i++) {
        if (isPrime(*(arr + i))) {
            printf("%d ", *(arr + i));
            sum += *(arr + i);
        }
    }
    printf("\nSum of primes = %d\n", sum);

    free(arr);
    return 0;
}
