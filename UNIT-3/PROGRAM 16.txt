#include <stdio.h>
#include <string.h>

int isPalindrome(char *str) {
    char *left  = str;
    char *right = str + strlen(str) - 1;
    while (left < right) {
        if (*left != *right) return 0;
        left++;
        right--;
    }
    return 1;
}

int main() {
    char str[100];

    printf("Enter a string: ");
    fgets(str, sizeof(str), stdin);
    str[strcspn(str, "\n")] = '\0';

    if (isPalindrome(str))
        printf("\"%s\" is a PALINDROME\n", str);
    else
        printf("\"%s\" is NOT a palindrome\n", str);

    return 0;
}
