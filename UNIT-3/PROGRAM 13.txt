#include <stdio.h>
#include <string.h>

int main() {
    char str[100], rev[100];
    int i, j = 0;

    printf("Enter a string: ");
    fgets(str, sizeof(str), stdin);
    str[strcspn(str, "\n")] = '\0';

    int len = strlen(str);

    /* Build reversed string manually */
    for (i = len - 1; i >= 0; i--)
        rev[j++] = str[i];
    rev[j] = '\0';

    printf("Original : %s\n", str);
    printf("Reversed : %s\n", rev);

    /* In-place reverse using two pointers */
    char *left  = str;
    char *right = str + len - 1;
    while (left < right) {
        char temp = *left;
        *left++   = *right;
        *right--  = temp;
    }
    printf("In-place : %s\n", str);

    return 0;
}
