#include <stdio.h>
#include <string.h>

int main() {
    char s1[100], s2[100];

    printf("Enter first string : ");
    fgets(s1, sizeof(s1), stdin);
    s1[strcspn(s1, "\n")] = '\0';

    printf("Enter second string: ");
    fgets(s2, sizeof(s2), stdin);
    s2[strcspn(s2, "\n")] = '\0';

    /* With strcmp */
    if (strcmp(s1, s2) == 0)
        printf("\n(strcmp) Strings are SAME\n");
    else
        printf("\n(strcmp) Strings are DIFFERENT\n");

    /* Without strcmp — character by character */
    int i = 0, same = 1;
    while (s1[i] != '\0' && s2[i] != '\0') {
        if (s1[i] != s2[i]) { same = 0; break; }
        i++;
    }
    if (s1[i] != s2[i]) same = 0;

    printf("(manual)  Strings are %s\n",
           same ? "SAME" : "DIFFERENT");

    return 0;
}
