#include <stdio.h>
#include <string.h>

int main() {
    char s1[100], s2[50], result[150];
    int i, j;

    printf("Enter first string : ");
    fgets(s1, sizeof(s1), stdin);
    s1[strcspn(s1, "\n")] = '\0';

    printf("Enter second string: ");
    fgets(s2, sizeof(s2), stdin);
    s2[strcspn(s2, "\n")] = '\0';

    /* With library function */
    strcpy(result, s1);
    strcat(result, s2);
    printf("\nConcatenated (strcat): %s\n", result);

    /* Without library function */
    i = 0; j = 0;
    while (s1[i]) result[i] = s1[i++];
    while (s2[j]) result[i++] = s2[j++];
    result[i] = '\0';
    printf("Concatenated (manual): %s\n", result);

    return 0;
}
