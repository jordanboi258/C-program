#include <stdio.h>
#define N 5

struct student {
    int   rollno;
    char  name[20];
    char  department[40];
    int   year_of_joining;
    float score;
};

void sortByScoreDesc(struct student s[], int n) {
    struct student temp;
    int i, j;
    for (i = 0; i < n - 1; i++)
        for (j = 0; j < n - 1 - i; j++)
            if (s[j].score < s[j+1].score) {
                temp   = s[j];
                s[j]   = s[j+1];
                s[j+1] = temp;
            }
}

void displayAll(struct student s[], int n) {
    int i;
    printf("\n%-5s %-20s %-10s\n", "Roll", "Name", "Score");
    printf("--------------------------------------\n");
    for (i = 0; i < n; i++)
        printf("%-5d %-20s %-10.2f\n",
               s[i].rollno, s[i].name, s[i].score);
}

int main() {
    struct student s[N] = {
        {3, "Zeynep",  "CSE", 2022, 78.5},
        {1, "Arjun",   "ECE", 2021, 91.0},
        {5, "Meera",   "IT",  2023, 65.0},
        {2, "Bhavesh", "CSE", 2020, 83.5},
        {4, "Priya",   "ME",  2022, 72.0}
    };

    printf("Before sort:");
    displayAll(s, N);

    sortByScoreDesc(s, N);

    printf("\nAfter sort (highest score first):");
    displayAll(s, N);
    return 0;
}
