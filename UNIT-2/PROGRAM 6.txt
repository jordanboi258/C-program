#include <stdio.h>
#define N 5

struct student {
    int   rollno;
    char  name[20];
    char  department[40];
    int   year_of_joining;
    float score;
};

void printByYear(struct student s[], int n, int year) {
    int i, found = 0;
    printf("\nStudents who joined in %d:\n", year);
    printf("-------------------------------\n");
    for (i = 0; i < n; i++) {
        if (s[i].year_of_joining == year) {
            printf("  Roll: %-4d  Name: %-20s  Dept: %s\n",
                   s[i].rollno, s[i].name, s[i].department);
            found++;
        }
    }
    if (!found)
        printf("  No students joined in %d.\n", year);
    else
        printf("\nTotal: %d student(s)\n", found);
}

int main() {
    struct student s[N] = {
        {1, "Arjun",   "ECE", 2019, 91.0},
        {2, "Bhavesh", "CSE", 2020, 83.5},
        {3, "Meera",   "IT",  2019, 65.0},
        {4, "Priya",   "ME",  2022, 72.0},
        {5, "Zeynep",  "CSE", 2019, 78.5}
    };

    printByYear(s, N, 2019);
    return 0;
}
