#include <stdio.h>

int main() {
    FILE *fp, *fodd, *feven;
    int i, num;

    fp = fopen("DATA.txt", "w");
    for(i=1;i<=10;i++)
        fprintf(fp,"%d ",i);
    fclose(fp);

    fp = fopen("DATA.txt","r");
    fodd = fopen("ODD.txt","w");
    feven = fopen("EVEN.txt","w");

    while(fscanf(fp,"%d",&num)!=EOF) {
        if(num%2==0)
            fprintf(feven,"%d ",num);
        else
            fprintf(fodd,"%d ",num);
    }

    fclose(fp); fclose(fodd); fclose(feven);
    return 0;
}
