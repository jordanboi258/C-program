#include <stdio.h>
#include <stdlib.h>

int main() {
    FILE *fp;
    char ch;

    // ── Write ──────────────────────────────────────────
    fp = fopen("sample.txt", "w");

    if (fp == NULL) {
        printf("Error opening file for writing!\n");
        exit(1);
    }

    fprintf(fp, "Hello, World!\n");
    fprintf(fp, "This is a C file I/O demo.\n");
    fprintf(fp, "Writing multiple lines to a file.\n");
    fprintf(fp, "C makes file handling powerful and efficient!\n");

    fclose(fp);
    printf("✅ Content written to file successfully!\n\n");
    printf("----------------------------------------\n");

    // ── Read ───────────────────────────────────────────
    fp = fopen("sample.txt", "r");

    if (fp == NULL) {
        printf("Error opening file for reading!\n");
        exit(1);
    }

    printf("📄 Content read from file:\n");
    printf("----------------------------------------\n");

    while ((ch = fgetc(fp)) != EOF) {
        putchar(ch);
    }

    fclose(fp);

    return 0;
}
