#include <stdio.h>

int main() {
    FILE *fp;
    char ch;

    fp = fopen("file.txt","r");

    fseek(fp,2,SEEK_SET);
    ch = fgetc(fp);
    printf("After fseek: %c\n", ch);

    printf("Position: %ld\n", ftell(fp));

    rewind(fp);
    ch = fgetc(fp);
    printf("After rewind: %c\n", ch);

    fclose(fp);
}
