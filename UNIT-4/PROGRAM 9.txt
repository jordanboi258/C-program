#include <stdio.h>
#include <ctype.h>

int main() {
    FILE *fp;
    char ch;
    int upper=0, lower=0, digit=0, vowel=0, special=0;

    fp = fopen("test.txt","r");

    while((ch=fgetc(fp))!=EOF) {
        if(isupper(ch)) upper++;
        else if(islower(ch)) lower++;
        else if(isdigit(ch)) digit++;
        else special++;

        if(ch=='a'||ch=='e'||ch=='i'||ch=='o'||ch=='u'||
           ch=='A'||ch=='E'||ch=='I'||ch=='O'||ch=='U')
           vowel++;
    }

    printf("Upper:%d Lower:%d Digit:%d Vowel:%d Special:%d",
           upper,lower,digit,vowel,special);

    fclose(fp);
}
