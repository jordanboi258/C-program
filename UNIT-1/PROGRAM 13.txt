#include <stdio.h>
#include <string.h>
int main() {
    char str[100];
    int i = 0, len = 0;
    printf("Enter a string: ");
    fgets(str, sizeof(str), stdin);

    /* With string function */
    printf("Length (strlen): %lu\n", strlen(str) - 1);

    /* Without string function */
    while (str[i] != '\0' && str[i] != '\n') { len++; i++; }
    printf("Length (manual): %d\n", len);
    return 0;
}
