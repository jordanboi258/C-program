#include <stdio.h>
int main() {
    char str[200];
    int i, spaces = 0, vowels = 0;
    printf("Enter a string: ");
    fgets(str, sizeof(str), stdin);
    for (i = 0; str[i] != '\0'; i++) {
        char c = str[i];
        if (c == ' ') spaces++;
        if (c=='a'||c=='e'||c=='i'||c=='o'||c=='u'||
            c=='A'||c=='E'||c=='I'||c=='O'||c=='U')
            vowels++;
    }
    printf("Spaces: %d\nVowels: %d\n", spaces, vowels);
    return 0;
}
