#include <stdio.h>
int main() {
    int n, i;
    printf("Enter number of elements: ");
    scanf("%d", &n);
    int src[n], dst[n];
    printf("Enter elements:\n");
    for (i = 0; i < n; i++) scanf("%d", &src[i]);
    for (i = 0; i < n; i++) dst[i] = src[i];
    printf("Copied array: ");
    for (i = 0; i < n; i++) printf("%d ", dst[i]);
    return 0;
}
