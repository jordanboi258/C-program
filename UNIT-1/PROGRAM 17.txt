#include <stdio.h>
#include <string.h>
int main() {
    char str[100];
    int i, len, isPalin = 1;
    printf("Enter a string: ");
    fgets(str, sizeof(str), stdin);
    str[strcspn(str,"\n")] = '\0';
    len = strlen(str);
    for (i = 0; i < len / 2; i++) {
        if (str[i] != str[len - 1 - i]) {
            isPalin = 0;
            break;
        }
    }
    printf("\"%s\" is %s\n", str, isPalin ? "a PALINDROME" : "NOT a palindrome");
    return 0;
}
